Hoisting

Variables and functions declarations are hoisted on top of the scope!

(function () {

	console.log(foo);
	foo = getValue();

	var foo;
	console.log(foo);

	function getValue() {
		return 1 + 1;
	}

}());
		
(function () {
	var foo;
	function getValue() {
		return 1 + 1;
	}

	console.log(foo);
	foo = getValue();
	console.log(foo);
}());
		
var sum = function (arg1, arg2) {
	return arg1 + arg2;
};

sum(5, 4); // 9

var sumAll = function () {
	var sum = 0;
	for (var i = 0; i < arguments.length; i++) {
		sum += arguments[i];
	}
	return sum;
};

Adv. functions

Function context

"this" value

var obj {
	method: function () {
	    console.log(this);
	}
}
obj.method(); // obj
		
var func = function () {
	console.log(this);
};

func(); // window (global object);

this value is dynamic - depends on how the function is called

obj.method(); // obj
var func = obj.method;

func(); // window

every function has it's own context

var obj = {
	method: function () {
	    console.log(this);

	    (function () {
	        console.log(this);
	    }());
	}
}
obj.method(); // obj; window;

Context can be changed

var obj = {}, func = function () {
	console.log(this);
};

func(); // window
func.apply(obj); //obj
func.call(obj); //obj

func(); //window

var func2 = func.bind(obj);
func2(); // obj
func2(); // obj

Common problems

var form = {
	addSubmitAction: function () {
        // ...
        formEl.addEventListener('submit', function () {
             this.validateData(); // !
        });
	},
	validateData: function () {
        //fancy validation here
	}
}

Solution one

var form = {
    addSubmitAction: function () {
        // ...
        formEl.addEventListener('submit', function () {
            this.validateData();
        }.bind(this));
	},
	validateData: function () {
	    //fancy validation here
	}
}

Solution one

var form = {
	addSubmitAction: function () {
		// ...
		formEl.addEventListener('submit', this.validateData.bind(this));
	},
	validateData: function () {
		//fancy validation here
		this.getData(); //...
	}
}

Solution two

var form = {
	addSubmitAction: function () {
		var me = this;
		// ...
		formEl.addEventListener('submit', function () {
            me.validateData();
		});
	},
	validateData: function () {
		//fancy validation here
	}
}
